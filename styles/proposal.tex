        \newacronymsaved{#1}{#2}{#3}

                            name={#2},

                            plural={⟨#2⟩\glspluralsuffix},
                            firstplural={⟨#3⟩\glspluralsuffix\space (⟨#2⟩\glspluralsuffix)},
                            see=[Glossary:]{#1-gls}}
        \newglossaryentry{#1-gls}{name={#2}, description={#4. \textit{See pgs}}}
